前提

修改配置redis.conf

## 注释掉，能监听到远程主机的连接
## bind 127.0.0.1
## 关闭保护模式，保护模式下，远程主机无法访问redis
protected-mode no
## 身份验证密码
requriepass root

Jedis

依赖

<dependency>
    <groupId>redis.clients</groupId>
    <artifactId>jedis</artifactId>
    <version>3.2.0</version>
    <type>jar</type>
    <scope>compile</scope>
</dependency>

代码

public class TestRedisClient {

    static  final String HOST = "";
    static  final Integer PORT = 6379;
    static  final String AUTH = "root";

    // Jedis 直连，每次都会新建TCP连接
    @Test
    public void jedisTest(){
         // 要记得的管理资源哦  jedis.close()
        try(Jedis jedis = new Jedis(HOST)){  // HOST 为redis 主机地址，端口默认为6379
            jedis.auth(AUTH); // 授权
            jedis.select(0); // scheme
            String s = jedis.get("test:hello");
            jedis.set("test:redis", "hello");
            Set<String> keys = jedis.keys("*");
            keys.forEach(System.out::println);
            System.out.println(s);
        }
    }

    // JedisPool连接
    @Test
    public  void jedisPoolTest(){
        // 创建连接池
        // 可通过final GenericObjectPoolConfig config = new GenericObjectPoolConfig();  配置连接池
        JedisPool jedisPool = new JedisPool(HOST, PORT);
        // 获得连接
        try (Jedis jedis = jedisPool.getResource()){
            // 授权
            jedis.auth(AUTH);
            // 查询
            String s = jedis.get("test:hello"); 
            jedis.set("test:redis","hello");
            Set<String> keys = jedis.keys("*");
            keys.forEach(System.out::println);
            System.out.println(s);
        }
    }
}

Lettuce

依赖

<dependency>
    <groupId>io.lettuce</groupId>
    <artifactId>lettuce-core</artifactId>
    <version>5.2.2.RELEASE</version>
</dependency>

代码

public class TestRedisClient {

    static final String HOST = "";
    static final Integer PORT = 6379;
    static final String AUTH = "root";

    @Test
    public void lettuceTest() {
        String url = "redis://" + AUTH + "@" + HOST+":"+PORT;
        RedisClient redisClient = RedisClient.create(url);
        StatefulRedisConnection<String, String> connect = redisClient.connect();
        RedisCommands<String, String> commands = connect.sync();
        final List<String> keys = commands.keys("*");
        keys.forEach(System.out::println);
    }
}

SpringBoot

  1. SpringBoot 1.x底层使用的Jedis，2.x 之后选用 Lettuce
  2. springboot 使用redis 十分简单只要配置属性即可
  3. 自动配置原理查看 org.springframework.boot.autoconfigure.cache.CacheAutoConfiguration类
  4. 学成工具类可参考springboot redis 项目实战 完整篇 RedisUtil.java 的实现

测试类

@RunWith(SpringJUnit4ClassRunner.class)
@SpringBootTest
public class TestSpringBootRedis {
    @Autowired
    RedisTemplate redisTemplate;

    // 添加k-v序列化
    @Before
    public void testBefore() {
        redisTemplate.setKeySerializer(new StringRedisSerializer());
        Jackson2JsonRedisSerializer jacksonSeial = new Jackson2JsonRedisSerializer(Object.class);
        ObjectMapper om = new ObjectMapper();
        // 指定要序列化的域，field,get和set,以及修饰符范围，ANY是都有包括private和public
        om.setVisibility(PropertyAccessor.ALL, JsonAutoDetect.Visibility.ANY);
        jacksonSeial.setObjectMapper(om);
        redisTemplate.setValueSerializer(jacksonSeial);
        // 设置hash key 和value序列化模式
        redisTemplate.setHashKeySerializer(new StringRedisSerializer());
        redisTemplate.setHashValueSerializer(jacksonSeial);
    }

    @Test
    @Ignore
    public void testRedisConnect() {
        final Object haha = redisTemplate.opsForValue().get("haha");
        redisTemplate.opsForValue().set("test:redis:springboot", "redis");
        final Object test = redisTemplate.opsForValue().get("test:redis:springboot");
        System.out.println(haha);
        System.out.println(test);
    }

    // 字符
    @Test
    public void testRedisString(){
        final ValueOperations opsForValue = redisTemplate.opsForValue();
        opsForValue.set("tom",buildUser("tom"));
        System.out.println(opsForValue.get("tom"));

        System.out.println( redisTemplate.opsForValue().get("haha"));
    }

    // hash
    @Test
    public  void testRedisHash(){
        final String hashName ="test:hash:type";
        final HashOperations opsForHash = redisTemplate.opsForHash();
        opsForHash.put(hashName,"SH","上海");
        opsForHash.put(hashName,"ZH","深圳");
        System.out.println("======hget==============");
        System.out.println(opsForHash.get(hashName, "SH"));
        System.out.println("======hmget==============");
        opsForHash.multiGet(hashName, Arrays.asList("SH", "ZH")).forEach(System.out::println);
        System.out.println("======hvals==============");
        opsForHash.values(hashName).forEach(System.out::println);
        System.out.println("======hkeys==============");
        opsForHash.keys(hashName).forEach(System.out::println);
    }


    private User buildUser(String userName) {
        return User.builder()
                .name(userName)
                .id(new Random().nextLong())
                .age(20)
                .createTime(new Date()).build();
    }
}
